create or replace view v_games_per_level as 
select nov_level.level, (select count(*) from game
where game.level = nov_level.level) as anzahl
from nov_level
order by nov_level.level

create or replace view v_user as 
SELECT u.name, u.id, u.created, u.mobile, u.credits, u.level,
(select max(g.beginn) from game as g where g.user_id = u.id) as letztes, 
(select count(g.beginn) from game as g where g.user_id = u.id) as anzahl
from nov_user as u
order by u.id

create or replace view v_last_games as
select u.name, g.beginn, g.level, g.score
from game as g, nov_user as u
where g.user_id = u.id
order by g.id desc
