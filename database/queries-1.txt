select u.id, u.name,
(select max(ifnull(g.score, 0)) 
 from game as g
 where g.user_id = u.id
  and g.level = 1) as score
from nov_user as u
order by score desc


select u.id, u.name, nov_level.level, g.score, g.ende 
from nov_user as u
inner join game as g
on g.user_id = u.id
inner join nov_level
on g.level = nov_level.level
where g.score > 0
  and g.score = (select max(g1.score) 
 from game as g1
 where g1.user_id = u.id
  and g1.level = nov_level.level)
  and nov_level.level = 1
  order by g.score desc

select distinct (ip) ip from nov_log where substr(ip,3,1) = '.' or substr(ip,4,1) = '.' 
union
select distinct substr(ip,1,10) ip from nov_log where substr(ip,5,1) = ':'
order by 1

select ip from nov_log where substr(ip,1,10) in (select distinct substr(ip,1,10) ip from nov_log where substr(ip,3,1) = '.')
union
select ip from nov_log where substr(ip,1,11) in (select distinct substr(ip,1,11) ip from nov_log where substr(ip,4,1) = '.')
union
select ip from nov_log where substr(ip,1,10) in (select distinct substr(ip,1,10) ip from nov_log where substr(ip,5,1) = ':')
order by 1

create or replace view v_games_per_user as SELECT u.name, (select count(*) from game g where g.user_id = u.id) cnt
FROM nov_user u
where (select count(*) from game g where g.user_id = u.id) > 1
order by 2 desc
